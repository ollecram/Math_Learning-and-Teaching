\subsubsection{Ex 0.2 (1)}\label{sec:Ex_0.2.1}
% Ex 1 Statement
Let $S, T, T'$ be sets. Show that
$$S \cap (T \cup T') = (S \cap T) \cup (S \cap T')$$
If $T_1,\ldots,T_n$ are sets, show that
$$S \cap (T_1 \cup \cdots \cup T_n) = (S \cap T_1) \cup \cdots \cup (S \cap T_n)\,.$$

% Ex 1 Solution
\paragraph{Solution:}{
\begin{align*}
x \in \left[ S \cap (T \cup T') \right] &\equiv  (x \in S) \land [x \in (T \cup T')]\\
                                        &\equiv  (x \in S) \land [(x \in T) \lor (x \in T')] \\
                                        &\equiv  [x \in S \land (x \in T)] \lor [x \in S \land (x \in T')]\\
                                        &\equiv  [x \in (S \cap T)] \lor [x \in (S \cap T')]\\
                                        &\equiv  x \in [(S \cap T) \cup (S \cap T')]\,. \qed
\end{align*}
The case involving the union of more than two $T$'s can be proved along the same line.
}	% Solution paragraph end!

\subsubsection{Ex 0.2 (2)}\label{sec:Ex_0.2.2}
% Ex 2 Statement
Show that the equalities of Exercise 1 remain true if the intersection and union signs $\cap$ and $\cup$ are interchanged.

% Ex 2 Solution
\paragraph{Solution:}{
The \tit{interchange} of \tit{intersection} and \tit{union} signs ($\cap$ and $\cup$) in the equality \tit{statement}
just causes the \tit{interchange} of the logical $\land$ (AND) and logical $\lor$ (OR) symbols:
\begin{align*}
x \in \left[ S \cup (T \cap T') \right] &\equiv  (x \in S) \lor [x \in (T \cap T')]\\
                                        &\equiv  (x \in S) \lor [(x \in T) \land (x \in T')] \\
                                        &\equiv  [x \in S \lor (x \in T)] \land [x \in S \lor (x \in T')]\\
                                        &\equiv  [x \in (S \cup T)] \land [x \in (S \cup T')]\\
                                        &\equiv  x \in [(S \cup T) \cap (S \cup T')]\,. \qed
\end{align*}
}	% Solution paragraph end!

\subsubsection{Ex 0.2 (3)}\label{sec:Ex_0.2.3}
% Ex 3 Statement
Let $A, B$ be subsets of a set $S$. Denote by $\cC_S(A)$ the complement of $A$ in $S$. Show that the complement of the intersection is the union of the complements, i.e.:
$$\cC_S(A \cap B) = \cC_S(A) \cup \cC_S(B) \text{   and   } \cC_S(A \cup B) = \cC_S(A) \cap \cC_S(B)\,.$$ 

% Ex 3 Solution
\paragraph{Solution:}{

\begin{align*}
x \in \left[ \cC_S(A \cap B) \right] &\equiv  [x \in \cC_S(A)] \lor [x \in \cC_S(B)] \\ 
                                     &\equiv  x \in [\cC_S(A) \cup \cC_S(B)]\,. \qed
\end{align*}

}	% Solution paragraph end!

\subsubsection{Ex 0.2 (4)}\label{sec:Ex_0.2.4}
% Ex 4 Statement
If $X, Y, Z$ are sets, show that
\begin{align*}
(X \cup Y) \times Z &= (X \times Z) \cup (Y \times Z)\,,\\ 
(X \cap Y) \times Z &= (X \times Z) \cap (Y \times Z)\,.\\ 
\end{align*}

% Ex 4 Solution
\paragraph{Solution:}{
Both $(X \cup Y) \times Z$ and $(X \times Z) \cup (Y \times Z)$ define the same set, namely the set of all $(k,z)$ pairs of which $k$ is member of $X$ \tit{OR}, \tit{inclusively}, it is member of $Y$. That is: $k \in (X \cup Y)$, while $z \in Z$. \\

Both $(X \cap Y) \times Z$ and $(X \times Z) \cap (Y \times Z)$ define the same set, namely the set of all $(k,z)$ pairs of which $k$ is a member of both sets $X$ and $Y$, i.e: $k \in (X \cap Y)$, while $z \in Z$. $Z$ itself may have a nonempty intersection with $X$, $Y$ or both, but that is irrelevant, as it only determines the range of possible values of the second component of the $(k, z)$ pair.
}	% Solution paragraph end!

\subsubsection{Ex 0.2 (5)}\label{sec:Ex_0.2.5}
% Ex 5 Statement
Let $f : S \rightarrow T$ be a mapping, and let $Y, Z$ be subsets of $T$. Show that 
\begin{align*}
f^{-1}(Y \cap Z) &= f^{-1}(Y) \cap f^{-1}(Z)\,,\\ 
f^{-1}(Y \cup Z) &= f^{-1}(Y) \cup f^{-1}(Z)\,.\\ 
\end{align*}
% Ex 5 Solution A

\paragraph{Solution:}{
Here we exhibit a \tbi{proof by contradiction}.\\

\tbf{Part I} : proving that $f^{-1}(Y \cap Z) = f^{-1}(Y) \cap f^{-1}(Z)\,.$\\
\tbi{If} the assertion is \tbi{false}, then $\exists x$ such that
\begin{enumerate}[i)]
\item $(x \in Y) \land (x \in Z)$ \tbi{and}
\item $\left[ \left( f^{-1}(x) \notin f^{-1}(Y) \right) \lor \left( f^{-1}(x) \notin f^{-1}(Z) \right) \right]\,.$
\end{enumerate}
But the assumption about $x$ made at \tbf{i)} ($\land$ : \quotes{\tbi{and}}) \tit{contradicts} the statement at \tbf{ii)}
(where $\lor$ : \quotes{\tbi{or}}). In words: \quotes{If $x$ is a member of both $Y$ and $Z$} its inverse image $f^{-1}(x)$ must be a member of the inverse images of both sets: $Y$ and $Z$. Therefore, an $x$ satisfying both \tbf{i)} and \tbf{ii)} \tbi{cannot exist}.\\ 

\tbf{Part II} : proving that $f^{-1}(Y \cup Z) = f^{-1}(Y) \cup f^{-1}(Z)\,.$\\
\tbi{If} the assertion is \tbi{false}, then $\exists x$ such that
\begin{enumerate}[i)]
\item $(x \in Y) \lor (x \in Z)$ \tbi{and}
\item $\left[ \left( f^{-1}(x) \notin f^{-1}(Y) \right) \land \left( f^{-1}(x) \notin f^{-1}(Z) \right) \right]\,.$
\end{enumerate}
But the assumption about $x$ made at \tbf{i)} ($\lor$ : \quotes{\tbi{or}}) \tit{contradicts} the statement at \tbf{ii)}
(where $\land$ : \quotes{\tbi{and}}). In words: \quotes{If $x$ is a member of $Y$, $Z$ \tit{or both}} its inverse image $f^{-1}(x)$ must be a member of the inverse images of \tit{at least one} among sets: $Y$ and $Z$. Therefore, an $x$ satisfying both \tbf{i)} and \tbf{ii)} \tbi{cannot exist}.\\ 
}	% Solution 5A paragraph end!

\paragraph{Alternative solutions:}{
The above paragraphs provide proofs by contradiction. Here we propose an alternative (more \quotes{\tit{constructive}}) type of proofs.\\

Assuming $f : S \rightarrow T$ to be \tit{surjective} is not a serious restriction\footnote{Otherwise take $T' = f(S)$, i.e. assume $T$ to be the codomain of $f$.}, yet it avoids some annoying details in the proof. Henceforth, assume that for any $y \in T$ there is $x \in S$ such that $f(x) = y$. We start by proving an \tbi{intermediate result}, namely that $f^{-1}$ \tit{maps partitions into partitions}\footnote{Or, more generally: that it \tit{maps nonintersecting sets to nonintersecting sets}.}.

Let $\{T_1, \ldots , T_n\}$ be \tit{subsets} of $T$ implementing an arbitrary\\ \tbi{partition} of the latter, so that  
$$ \bigcup\limits_{i=1}^{n} T_i = T\, \text{ \tbi{and}, for any } (i,j)\, \text{ with } i \neq j\,, T_i \cap T_j = \emptyset\,.$$
We claim that $\{S_1, \ldots , S_n\}$, where   $S_i = f^{-1}(T_i)$, is a \tbi{partition} of $S$:
$$ \bigcup\limits_{i=1}^{n} S_i = S\, \text{ \tbi{and}, for any } (i,j)\, \text{ with } i \neq j\,, S_i \cap S_j = \emptyset\,.$$
\tbf{Proof}:\\
Recap of our definitions: $S_i = f^{-1}(T_i)\;$ \text{ and } $T_i = f(S_i)\,.$\\
\tbi{Step 1:} We prove that $\:\forall i, \text{ and } j \neq i \;  S_i \cap S_j =  f^{-1}(T_i) \cap f^{-1}(T_j) = \emptyset\,,$\\ 
by showing that assuming the contrary, (that $\exists x \in (S_i \cap S_j)\,$), leads to a contradiction. Indeed:\\
$\exists x \in (S_i \cap S_j) \implies \exists f(x) \in (T_i \cap T_j)\:$ but $T_i$ and $T_j$ are \tit{members} of \tbi{nonintersecting subsets} of $T$, hence $(T_i \cap T_j) = \emptyset \qed$.\\

Now, if $Y, Z$ are any two subsets of $T$, let consider their \tbi{union} $Y \cup Z$, their \tbi{intersection} $Y \cap Z$ and the set differences $Y\setminus Z$ and $Z\setminus Y$. Clearly, $Y \cup Z$ can be considered as the union of \tbi{three} \tit{nonintersecting} sets: 
$$ Y \cup Z = (Y\setminus Z) \cup (Y \cap Z) \cup (Z\setminus Y)\,.$$
By applying  to $ Y\cup Z$ the above result (that $f^{-1}$ maps nonintersecting sets to nonintersecting sets), we obtain\footnote{It should be evident, but we have not actually proved, that the last (third) equality follows from the previous one.}
\begin{align*}
f^{-1}(Y \cup Z) &= f^{-1}\left[ (Y\setminus Z) \cup (Y \cap Z) \cup (Z\setminus Y) \right] \,,\\
                 &= f^{-1}(Y\setminus Z) \cup f^{-1}(Y \cap Z) \cup f^{-1}(Z \setminus Y)\,, \\
                 &= f^{-1}(Y) \cup f^{-1}(Z)\,. \qed\\
\end{align*}
Let now turn to $ Y\cap Z$.  For all $x \in (Y \cap Z)$  ($Y, Z$ subsets of $T$)
\begin{align*}
x \in Y &\implies f^{-1}(x) \in f^{-1}(Y)\,, \text{while}\\
x \in Z &\implies f^{-1}(x) \in f^{-1}(Z)\,, \text{hence}\\
x \in (Y\cap Z) &\implies f^{-1}(x) \in (f^{-1}(Y) \cap f^{-1}(Z))\,. \qed 
\end{align*}

}	% Solution 5B paragraph end!


\subsubsection{Ex 0.2 (6)}\label{sec:Ex_0.2.6}
% Ex 6 Statement
Let $S, T, U$ be sets, and let $f : S \rightarrow T$ and $g : T \rightarrow U$ be mappings.
\begin{enumerate}[(a)]
\item If $g, f$ are injective, show that $g \circ f$ is injective. 
\item If $f, g$ are surjective, show that $g \circ f$ is surjective.
\end{enumerate}
% Ex 6 Solution
\paragraph{Solution:}{
%	START part (a)
Part (a)\\
\begin{align*}
\text{Let } x_1, x_2 \in S\,,\: y_1 &= f(x_1)\,,\:  y_2 = f(x_2) \in T\,,\\
                                z_1 &= g(y_1)\,,\:  z_2 = g(y_2) \in U\,.\\
(x_1 \neq x_2)  &\implies  (y_1 \neq y_2)\qquad \text{ (due to } f \text{ being injective.)}\\   
(y_1 \neq y_2)  &\implies  (z_1 \neq z_2)\qquad \text{ (due to } g \text{ being injective.)}\\
\text{Therefore}\\
(x_1 \neq x_2)  &\implies  (z_1 \neq z_2)\:\: \text{hence }\:\: g \circ f \text{ is injective.)} \qed   
\end{align*}

%   START part (b) 
Part (b)\\
\begin{align*}
g \text{ surjective } &\implies \qquad \forall z \in U, \: \exists y \in T \text{ such that } g(y) = z\,\\
f \text{ surjective } &\implies \qquad \forall y \in T, \: \exists x \in S \text{ such that } f(x) = y\,\\
\text{Therefore}\\
f,g \text{ \tbi{both} surjective } &\implies \forall z \in U, \: \exists x \in S \text{ such that } (g \circ f)(x) = z\,\\
\text{hence }\:\: g \circ f \text{ is surjective.)} \qed 
\end{align*}

\subsubsection{Ex 0.2 (7)}\label{sec:Ex_0.2.7}
% Ex 7 Statement
Let $S, T$ be sets, and $f : S \rightarrow T$ a mapping.
Show that $f$ is bijective if and only if $f$ has an inverse mapping.

% Ex 7 Solution
\paragraph{Solution:}{
For an inverse mapping $f^{-1} : T \rightarrow S$ to exist 
\begin{itemize}
\item Any element $y \in T$ must be the image of some $x \in S$ ($f$ \tit{surjective});    
\item No  element $y \in T$ can be the image of more than a single element $x \in S$ ($f$ \tit{injective}).
\end{itemize}

It is thus a \tbi{necessary} condition for $f^{-1}$ to exist that $f$ be a bijective function.
That is also a \tbi{sufficient} condition, because the inverse image of any element $y \in T$ is 
just the \tit{unique} element $x \in S$ such that $f(x) = y\,.\qed$
}

\subsubsection{Ex 0.3 (1)}\label{sec:Ex_0.3.1}
% Ex 1 statement
Prove the following statements for all positive integers:
\begin{enumerate}[(a)]
\item $1 + 3 + 5 + \cdots + (2n - 1) = n^2$
\item $1^2 + 2^2 + \cdots + n^2 = n (n+1)(2n + 1)/6$
\item $1^3 + 2^3 + \cdots + n^3 = [n (n+1)/2]^2$
\end{enumerate}

%
% Ex 1 solution
%
\paragraph{Solution:}{
\begin{enumerate}[(a)]
\item $\sum_{i=1}^{n} (2n-1) = 2 \sum_{i=1}^{n} i - \sum_{i=1}^{n} 1 = 2 \frac{n(n+1)}{2} - n = n(n+1) - n = n^2\,.$ 
\item The statement is true for $n = 1\,: 1^2 = 1 \cdot 2 \cdot 3/6 = 6/6$. By induction, it is enough to prove that \tbi{if} the sum of the squares $i^2$ of the first $n$ integers equals $n (n+1)(2n + 1)/6$ \tbi{then} by adding $(n+1)^2$ to this sum one obtains $m (m+1)(2m + 1)/6$, with $m=n+1$. Indeed both expressions reduce to $2 n^3 + 9 n^2 + 13 n + 6$, hence the identity is proved.
\item The statement is true for $n = 1\,: 1^3 = [1\cdot(1+1)/2)]^2 = 1$. By induction, it is enough to prove that \tbi{if} the sum of the cubes $i^3$ of the first $n$ integers equals $[n (n+1)/2]^2$ \tbi{then} by adding $(n+1)^3$ to this sum one obtains $[m (m+1)/2]^2$, with $m=n+1$. Indeed both expressions reduce to $n^4+ 6 n^3 + 13 n^2 + 12 n + 4$, hence the identity is proved.
\end{enumerate}
}

\subsubsection{Ex 0.3 (2)}\label{sec:Ex_0.3.2}
% Ex 2 statement
Prove that for all numbers $x \neq 1$,
$$ (1 + x)(1 + x^2)(1 + x^4)\cdots(1 + x^{2^{n}}) = \frac{1 - x^{2^{n + 1}}}{1 - x}$$

%
% Ex 2 solution
%
\paragraph{Solution:}{
We are facing a product whose first term can be written as $1 + x^{2^{0}} = 1 + x^{1} = (1+x)\,$ 
while the last term is $1 + x^{2^{n}}$.\\ We must prove that

$$P(n, x) \equiv \prod_{k=0}^{n} (1 + x^{2^{k}}) = F(n, x) \equiv \frac{1 - x^{2^{n + 1}}}{1 - x}\,.$$

We start verifying that $P(0, x) \equiv F(0, x)\,$:

$$   P(0, x) = (1 + x^{2^{0}}) = (1 + x) \equiv F(0, x) = \frac{1 - x^{2}}{1 - x} = \frac{(1 + x)(1 - x)}{1 - x}\,,$$

then we continue with a proof by induction by showing that $$P(n, x) \equiv F(n, x) \implies P(n+1, x) \equiv F(n+1, x)\,:$$ 
\begin{align*}
P(n+1, x) &= P(n, x) \cdot (1 + x^{2^{(n+1)}})\\
          &= \frac{(1 - x^{2^{(n + 1)}})}{1 - x} \cdot (1 + x^{2^{(n+1)}})\\
          &= \frac{  (1 - x^{2^{(n + 1)}})  \cdot (1 + x^{2^{(n+1)}}) }{1 - x} \\
          &= \frac{  \left(1 - {\left( x^{2^{(n + 1)}} \right)}^2 \right)  }{1 - x}\\
          &= \frac{  \left(1 - x^{2^{(n + 2)}} \right)  }{1 - x} \equiv F(n+1, x)\,\qed.
\end{align*}
}

\subsubsection{Ex 0.3 (3)}\label{sec:Ex_0.3.3}
% Ex 3 statement
Let $f: \N \rightarrow \N$ be a mapping such that $f(xy) = f(x) + f(y)$ for all $x, y$. 
Show that $f(a^n) = n f(a)$ for all $n \in \N\,.$ 
%
% Ex 3 solution
%
\paragraph{Solution:}{
We must prove that 
$$P(n, a) \equiv f(a^n) = n \cdot f(a) \equiv F(n, a)\,.$$

We start verifying that $P(1, a) \equiv F(1, a)\,$:
$$ P(1, a) = f(a^1) = 1 \cdot f(a)\,,$$
then we continue with a proof by induction by showing that $$P(n, a) \equiv F(n, a) \implies P(n+1, a) \equiv F(n+1, a)\,:$$  
\begin{align*}
P(n+1, a) &= f(a^{n+1}) = f(a \cdot a^n) \\
          &= f(a) + f(a^n) = f(a) + n f(a)\\
          &= (n+1) f(a) \equiv F(n+1, a)\, \qed
\end{align*}

}


\subsubsection{Ex 0.3 (4)}\label{sec:Ex_0.3.4}
% Ex 4 statement
$$\text{Let } \binom{n}{k} \text{ denote the binomial coefficient, } \binom{n}{k} = \frac{n!}{k!(n-k)!}\,,$$
%$$\binom{n}{k} = \frac{n!}{k!(n-k)!}\,,$$
where $n, k$ are integers $\geq 0$, $0 \leq k \leq n$, and $0!$ is defined to be $1$.
Also $n!$ is defined to be the product $1 \cdot 2 \cdot 3 \cdots n$. Prove the following  assertions
$$ (a)\: \binom{n}{k} = \binom{n}{n-k}\qquad (b)\: \binom{n}{k-1} + \binom{n}{k} = \binom{n+1}{k}\: \text{  ( for }k > 0\,\text{)}$$

%
% Ex 4 solution
%
\paragraph{Solution:}{
\begin{enumerate}[(a)]
\item $$ \binom{n}{n-k} = \frac{n!}{(n-k)! [n - (n-k)]!} = \frac{n!}{(n-k)! k!} = \binom{n}{k}\,\qed$$
\item
\begin{align*}
\binom{n}{k-1} + \binom{n}{k} &= \frac{n!}{(k-1)! (n-k+1)!} +  \frac{n!}{k! (n-k)!}\\
                              &= \frac{[(n + 1 - k) + k] n!}{k! (n + 1 - k)!}  = \frac{(n+1)\cdot n!}{k!(n + 1 - k)!}\\
                              &= \frac{(n + 1)!}{k! (n + 1 - k)!} = \binom{n+1}{k}\, \qed
\end{align*}
\end{enumerate}
}


\subsubsection{Ex 0.3 (5)}\label{sec:Ex_0.3.5}
% Ex 5 statement
\tit{Prove by induction that}
$$ (x + y)^n = \sum_{k=0}^n \binom{n}{k} x^k y ^{n-k}\,.$$
%
% Ex 5 solution
%
\paragraph{Solution:}{
The formula is true for $n = 1$, since by definition $0! = 1$, and $\binom{n}{0} = \binom{n}{n} = 1$. 
Assuming the formula to be true for $n \geq 1$, we must show that it is true for $n+1$ as well: 
\begin{align*}
(x + y)^{n+1} &= (x + y) \left[ \sum_{k=0}^n \binom{n}{k} x^k y ^{n-k}   \right]\\
              &= \sum_{k=0}^n \binom{n}{k} x^{k+1} y^{n-k} + \sum_{k=0}^n \binom{n}{k} x^k y ^{n+1-k}\\
              &= \sum_{k=1}^{n+1} \binom{n}{k-1} x^{k} y^{n+1-k} + \sum_{k=0}^n \binom{n}{k} x^k y ^{n+1-k}\\ 
              &= \sum_{k=1}^{n} \left[ \binom{n}{k-1} +  \binom{n}{k} \right] x^{k} y^{n+1-k} + \binom{n}{n} x^{n+1} + y^{n+1}\\
              &= \sum_{k=1}^{n} \binom{n+1}{k}  x^{k} y^{n+1-k}  + x^{n+1} + y^{n+1}\\
              &= \sum_{k=0}^{n+1} \binom{n+1}{k}  x^{k} y^{n+1-k} \qed
\end{align*}

\tbi{Note: }The sum of binomial coeficients within square parentheses \tbi{on the 4th row} is \tbi{replaced in the 5th row} according to the result of Exercise. 0.3.4 part (b).
